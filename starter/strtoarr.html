Strings to Array

A string can be converted to an array with the Split() method:

function myFunction() {

var str = "a,b,c,d,e,f,g";

var arr = str.split(",");

document.getElementById("id").innerHTML = arr[0];}

Example

var str = "Hello";

var arr = str.split("");

var text = "";

var i;

for (i = 0; i < arr.length; i++) {

text += arr[i] + "<"br">"

}

document.getElementById("demo").innerHTML = text;
